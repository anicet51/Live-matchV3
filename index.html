        let matchState = {

            title: '⚽ Score Football Live',

            homeTeam: '', awayTeam: '',

            homeLogoDataUrl: null, awayLogoDataUrl: null,

        let currentEventTeam = '';

        let currentEventType = '';


        // Charger les données au démarrage

        document.addEventListener('DOMContentLoaded', function() {

            loadData();

        document.getElementById('saveTitle').onclick = function() {

            const newTitle = document.getElementById('titleInput').value.trim();

            if (newTitle) {

                matchState.title = newTitle;

                document.getElementById('matchTitle').textContent = newTitle;

                saveData();

            }

        };


        document.getElementById('titleInput').onkeypress = function(e) {

            if (e.key === 'Enter') document.getElementById('saveTitle').click();

            handleLogoUpload(e, 'away');

        function handleLogoUpload(e, team) {

            const file = e.target.files[0];

            if (file && file.type.startsWith('image/')) {

                const reader = new FileReader();

                reader.onload = function(e) {

                    document.getElementById(team + 'Logo').innerHTML = 

                        '<img src="' + e.target.result + '" class="logo-img" alt="Logo ' + (team === 'home' ? 'Domicile' : 'Visiteur') + '">';

                    matchState[team + 'LogoDataUrl'] = e.target.result;

                    saveData();

                };

                reader.readAsDataURL(file);

            }

        }


        // Gestion des scores

        function changeScore(team, change) {

            const currentScore = matchState[team + 'Score'];

            const newScore = currentScore + change;

            if (newScore >= 0) {

                matchState[team + 'Score'] = newScore;

                updateDisplay();

                saveData();

            }

        }


        // Gestion du chrono

        function startMatch() {

            if (!matchState.isRunning) {

                if (!matchState.startTime) {

                    matchState.startTime = Date.now() - (matchState.currentTime * 1000);

                } else {

                    matchState.startTime = Date.now() - (matchState.currentTime * 1000);

                }

                matchState.isRunning = true;

                matchState.status = 'En cours';

                matchState.timerInterval = setInterval(updateTimer, 1000);


                document.getElementById('startBtn').disabled = true;

                document.getElementById('pauseBtn').disabled = false;


                updateDisplay();

                saveData();

            }

        }


        function pauseMatch() {

            if (matchState.isRunning) {

                matchState.isRunning = false;

                matchState.status = 'En pause';

                if (matchState.timerInterval) {

                    clearInterval(matchState.timerInterval);

                    matchState.timerInterval = null;

                }


                document.getElementById('startBtn').disabled = false;

                document.getElementById('pauseBtn').disabled = true;


                updateDisplay();

                saveData();

            }

        }


        function resetMatch() {

            if (matchState.timerInterval) {

                clearInterval(matchState.timerInterval);

                matchState.timerInterval = null;

            }


            matchState.startTime = null;

            matchState.currentTime = 0;

            matchState.isRunning = false;

            matchState.status = 'En attente';

            matchState.homeScore = 0;

            matchState.awayScore = 0;

            matchState.events = [];


            document.getElementById('startBtn').disabled = false;

            document.getElementById('pauseBtn').disabled = true;


            updateDisplay();

            updateEvents();

            updateTimeline();

            saveData();

        }


        function newMatch() {

            if (confirm('Effacer TOUT et commencer un nouveau match ?')) {

                localStorage.removeItem('footballMatchState');

                location.reload();

            }

        }


        function updateTimer() {

            if (matchState.isRunning && matchState.startTime) {

                matchState.currentTime = Math.floor((Date.now() - matchState.startTime) / 1000);

                updateDisplay();

            }

        }


        // Gestion de la modal d'événement

        function openEventModal(team, eventType) {

            currentEventTeam = team;

            currentEventType = eventType;


            const modal = document.getElementById('eventModal');

            const title = document.getElementById('modalTitle');

            const minuteInput = document.getElementById('eventMinute');

            const playerInput = document.getElementById('playerName');


            const eventLabels = {

                'goal': 'But',

                'yellow': 'Carton Jaune', 

                'red': 'Carton Rouge'

            };


            const teamLabel = team === 'home' ? 'Domicile' : 'Visiteur';

            title.textContent = eventLabels[eventType] + ' - ' + teamLabel;


            // Préremplir la minute avec le temps actuel

            const currentMinute = Math.floor(matchState.currentTime / 60);

            minuteInput.value = currentMinute;


            playerInput.value = '';

            playerInput.focus();


            modal.style.display = 'block';

        }


        function closeEventModal() {

            document.getElementById('eventModal').style.display = 'none';

            currentEventTeam = '';

            currentEventType = '';

        }


        function validateEvent() {

            const playerName = document.getElementById('playerName').value.trim();

            const minute = parseInt(document.getElementById('eventMinute').value) || 0;


            if (!playerName) {

                alert('Veuillez saisir le nom du joueur');

                return;

            }


            const event = {

                id: Date.now(),

                team: currentEventTeam,

                type: currentEventType,

                minute: minute,

                player: playerName,

                time: minute * 60, // Pour la timeline

                timestamp: Date.now()

            };


            if (currentEventType === 'goal') {

                matchState[currentEventTeam + 'Score']++;

            }


            matchState.events.push(event);


            updateDisplay();

            updateEvents();

            updateTimeline();

            saveData();

            closeEventModal();

        }


        // Événement simple sans modal (remplacements)

        function addSimpleEvent(team, eventType) {

            const event = {

                id: Date.now(),

                team: team,

                type: eventType,

                minute: Math.floor(matchState.currentTime / 60),

                time: matchState.currentTime,

                timestamp: Date.now()

            };


            matchState.events.push(event);

            updateDisplay();

            updateEvents();

            updateTimeline();

            saveData();

        }


        // Mise à jour de l'affichage

        function updateDisplay() {

            document.getElementById('homeScore').textContent = matchState.homeScore;

            document.getElementById('awayScore').textContent = matchState.awayScore;


            const minutes = Math.floor(matchState.currentTime / 60);

            const seconds = matchState.currentTime % 60;

            document.getElementById('chronoTime').textContent = 

                minutes.toString().padStart(2, '0') + ':' + seconds.toString().padStart(2, '0');


            document.getElementById('chronoStatus').textContent = matchState.status;


            // Sauvegarder les noms d'équipe

            const homeTeamInput = document.getElementById('homeTeam');

            const awayTeamInput = document.getElementById('awayTeam');

            if (homeTeamInput.value !== matchState.homeTeam) {

                matchState.homeTeam = homeTeamInput.value;

            }

            if (awayTeamInput.value !== matchState.awayTeam) {

                matchState.awayTeam = awayTeamInput.value;

            }

        }


        function updateEvents() {

            const homeEventsEl = document.getElementById('homeEvents');

            const awayEventsEl = document.getElementById('awayEvents');


            homeEventsEl.innerHTML = '';

            awayEventsEl.innerHTML = '';


            const sortedEvents = matchState.events.sort((a, b) => a.minute - b.minute);


            sortedEvents.forEach(event => {

                const eventEl = document.createElement('div');

                eventEl.className = 'event-item';


                const icon = getEventIcon(event.type);

                let content = '<span class="event-time">' + event.minute + "'</span>" +

                             '<span class="event-icon">' + icon + '</span>';


                if (event.player) {

                    content += '<span class="event-player">' + event.player + '</span>';

                } else {

                    content += '<span>' + getEventLabel(event.type) + '</span>';

                }


                eventEl.innerHTML = content;


                if (event.team === 'home') {

                    homeEventsEl.appendChild(eventEl);

                } else {

                    awayEventsEl.appendChild(eventEl);

                }

            });


            if (homeEventsEl.innerHTML === '') {

                homeEventsEl.innerHTML = '<div style="text-align: center; opacity: 0.7;">Aucun événement</div>';

            }

            if (awayEventsEl.innerHTML === '') {

                awayEventsEl.innerHTML = '<div style="text-align: center; opacity: 0.7;">Aucun événement</div>';

            }

        }


        function updateTimeline() {

            const container = document.getElementById('timelineContainer');


            if (matchState.events.length === 0) {

                container.innerHTML = '<div style="text-align: center; color: rgba(255,255,255,0.7); padding: 20px;">Aucun événement pour le moment</div>';

                return;

            }


            container.innerHTML = '';


            const sortedEvents = matchState.events.sort((a, b) => a.minute - b.minute);


            sortedEvents.forEach(event => {

                const item = document.createElement('div');

                item.className = 'timeline-item';


                const teamName = event.team === 'home' ? (matchState.homeTeam || 'Domicile') : (matchState.awayTeam || 'Visiteur');

                const eventIcon = getEventIcon(event.type);


                let eventInfo = getEventLabel(event.type);

                if (event.player) {

                    eventInfo += ' - ' + event.player;

                }


                item.innerHTML = 

                    '<span class="timeline-time">' + event.minute + "'</span>" +

                    '<span class="timeline-icon">' + eventIcon + '</span>' +

                    '<span class="timeline-info">' + teamName + ' : ' + eventInfo + '</span>';


                container.appendChild(item);

            });

        }


        function getEventIcon(eventType) {

            const icons = { 

                goal: '⚽', 

                yellow: '🟨', 

                red: '🟥', 

                sub: '🔄' 

            };

            return icons[eventType] || '📝';

        }


        function getEventLabel(eventType) {

            const labels = { 

                goal: 'But', 

                yellow: 'Carton Jaune', 

                red: 'Carton Rouge', 

                sub: 'Remplacement' 

            };

            return labels[eventType] || eventType;

        }

        function saveData() {

            try {

                const dataToSave = { ...matchState, timerInterval: null };

                localStorage.setItem('footballMatchState', JSON.stringify(dataToSave));

            } catch (error) {

                console.warn('Erreur sauvegarde:', error);

            }

        }


        function loadData() {

            try {

                const saved = localStorage.getItem('footballMatchState');

                if (saved) {

                    const savedState = JSON.parse(saved);

                    Object.assign(matchState, savedState);

                    matchState.timerInterval = null;


                    // Restaurer l'interface

                    if (matchState.title) {

                        document.getElementById('titleInput').value = matchState.title;

                        document.getElementById('matchTitle').textContent = matchState.title;

                    }

                    if (matchState.homeTeam) {

                        document.getElementById('homeTeam').value = matchState.homeTeam;

                    }

                    if (matchState.awayTeam) {

                        document.getElementById('awayTeam').value = matchState.awayTeam;

                    }

                    if (matchState.homeLogoDataUrl) {

                        document.getElementById('homeLogo').innerHTML = 

                            '<img src="' + matchState.homeLogoDataUrl + '" class="logo-img" alt="Logo Domicile">';

                    }

                    if (matchState.awayLogoDataUrl) {

                        document.getElementById('awayLogo').innerHTML = 

                            '<img src="' + matchState.awayLogoDataUrl + '" class="logo-img" alt="Logo Visiteur">';

                    }


                    updateEvents();

                    updateTimeline();

                }

            } catch (error) {

                console.warn('Erreur chargement:', error);

            }

        }


        // Sauvegarde automatique
